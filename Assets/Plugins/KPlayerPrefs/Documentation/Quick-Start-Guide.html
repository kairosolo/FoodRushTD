KPlayerPrefs - Quick Start Guide

A simple, encrypted replacement for Unity's PlayerPrefs with multi-profile support.

üöÄ What is KPlayerPrefs?

KPlayerPrefs is like Unity's PlayerPrefs, but better:

  • ‚úÖ Multiple player profiles (Player1, Player2, etc.)
  • ‚úÖ Automatic encryption (your data is safe)
  • ‚úÖ More data types (Vector3, Color, DateTime)
  • ‚úÖ Drop-in replacement (same familiar API)

‚ö° 5-Minute Setup

Step 1: Replace PlayerPrefs calls

// OLD - Unity PlayerPrefs
PlayerPrefs.SetInt("Level", 5);
int level = PlayerPrefs.GetInt("Level", 1);

// NEW - KPlayerPrefs
KPlayerPrefs.SetInt("Level", 5);           // Global settings
KPlayerPrefs.Profiles.SetInt("Level", 5);  // Player-specific data

Step 2: Add using statement

üìö Basic Usage

Global Data (Settings, Preferences)

// Save settings that apply to all players
KPlayerPrefs.SetFloat("MasterVolume", 0.8f);
KPlayerPrefs.SetString("Language", "English");
KPlayerPrefs.SetBool("ShowTutorial", true);

// Load settings
float volume = KPlayerPrefs.GetFloat("MasterVolume", 1.0f);
string language = KPlayerPrefs.GetString("Language", "English");
bool showTutorial = KPlayerPrefs.GetBool("ShowTutorial", true);

Profile Data (Player Progress)

// Create and switch to a player profile
KPlayerPrefs.Profiles.Create("Player1");
KPlayerPrefs.Profiles.SetActive("Player1");

// Save player-specific data
KPlayerPrefs.Profiles.SetString("PlayerName", "Hero");
KPlayerPrefs.Profiles.SetInt("Level", 25);
KPlayerPrefs.Profiles.SetFloat("Health", 100f);

// Load player data
string name = KPlayerPrefs.Profiles.GetString("PlayerName", "Unknown");
int level = KPlayerPrefs.Profiles.GetInt("Level", 1);
float health = KPlayerPrefs.Profiles.GetFloat("Health", 100f);

üéÆ Complete Example

using UnityEngine;
using KairosoloSystems;

public class GameManager : MonoBehaviour
{
    void Start()
    {
        // Set up global settings (shared by all players)
        KPlayerPrefs.SetFloat("MasterVolume", 0.8f);
        KPlayerPrefs.SetString("Language", "English");
        
        // Create player profile if it doesn't exist
        if (!KPlayerPrefs.Profiles.Exists("MainPlayer"))
        {
            KPlayerPrefs.Profiles.Create("MainPlayer");
        }
        
        // Switch to the player profile
        KPlayerPrefs.Profiles.SetActive("MainPlayer");
        
        // Load or set initial player data
        string playerName = KPlayerPrefs.Profiles.GetString("PlayerName", "New Player");
        int playerLevel = KPlayerPrefs.Profiles.GetInt("Level", 1);
        
        Debug.Log($"Welcome back, {playerName}! You're level {playerLevel}");
    }
    
    public void SavePlayerProgress(int newLevel, Vector3 position)
    {
        KPlayerPrefs.Profiles.SetInt("Level", newLevel);
        KPlayerPrefs.Profiles.SetVector3("LastPosition", position);
        KPlayerPrefs.Profiles.SetDateTime("LastPlayed", System.DateTime.Now);
        
        Debug.Log("Progress saved!");
    }
}

üé® Cool Features

Multiple Profiles

// Create different players
KPlayerPrefs.Profiles.Create("Player1");
KPlayerPrefs.Profiles.Create("Player2");

// Switch between them
KPlayerPrefs.Profiles.SetActive("Player1");
KPlayerPrefs.Profiles.SetString("PlayerName", "Alice");

KPlayerPrefs.Profiles.SetActive("Player2");
KPlayerPrefs.Profiles.SetString("PlayerName", "Bob");

// Direct access without switching (advanced)
KPlayerPrefs.Profiles.Profile("Player1").SetInt("Level", 10);
KPlayerPrefs.Profiles.Profile("Player2").SetInt("Level", 5);

Advanced Data Types

// Save positions, colors, dates
KPlayerPrefs.Profiles.SetVector3("LastPosition", transform.position);
KPlayerPrefs.Profiles.SetColor("PlayerColor", Color.red);
KPlayerPrefs.Profiles.SetDateTime("LastPlayed", System.DateTime.Now);

// Load them back
Vector3 lastPos = KPlayerPrefs.Profiles.GetVector3("LastPosition", Vector3.zero);
Color playerColor = KPlayerPrefs.Profiles.GetColor("PlayerColor", Color.white);
System.DateTime lastPlayed = KPlayerPrefs.Profiles.GetDateTime("LastPlayed", System.DateTime.Now);
