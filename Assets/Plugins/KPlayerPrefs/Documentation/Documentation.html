Advanced Features

Direct Profile Access

// Access specific profiles without switching the active profile
KPlayerPrefs.Profiles.Profile("Player1").SetString("PlayerName", "Hero");
KPlayerPrefs.Profiles.Profile("Player2").SetString("PlayerName", "Villain");
int player1Level = KPlayerPrefs.Profiles.Profile("Player1").GetInt("Level", 1);
int player2Level = KPlayerPrefs.Profiles.Profile("Player2").GetInt("Level", 1);

// Active profile remains unchanged
Debug.Log(KPlayerPrefs.Profiles.ActiveProfile); // Still whatever it was before

// Useful for comparing profiles or bulk operations
string[] allProfiles = KPlayerPrefs.Profiles.GetAll();
foreach(string profileName in allProfiles)
{
    var level = KPlayerPrefs.Profiles.Profile(profileName).GetInt("Level", 1);
    var lastPlayed = KPlayerPrefs.Profiles.Profile(profileName).GetDateTime("LastPlayed", DateTime.MinValue);
    Debug.Log($"{profileName}: Level {level}, Last played {(DateTime.Now - lastPlayed).Days} days ago");
}

// Mixed usage patterns
KPlayerPrefs.Profiles.SetActive("MainPlayer");                    // Set active for main gameplay
KPlayerPrefs.Profiles.SetInt("CurrentHP", 100);                   // Uses active profile
KPlayerPrefs.Profiles.Profile("Player2").SetInt("Level", 15);     // Direct access to other profile
KPlayerPrefs.Profiles.Profile("Player3").SetInt("Level", 30);     // Direct access to another profile
// Active profile is still "MainPlayer"

KPlayerPrefs is a drop-in replacement! Simply change your calls:

// OLD - Unity PlayerPrefs
PlayerPrefs.SetInt("Level", 5);
int level = PlayerPrefs.GetInt("Level", 1);

// NEW - KPlayerPrefs (Global Data)
KPlayerPrefs.SetInt("Level", 5);
int level = KPlayerPrefs.GetInt("Level", 1);

// NEW - KPlayerPrefs (Profile Data)
KPlayerPrefs.Profiles.SetInt("Level", 5);
int level = KPlayerPrefs.Profiles.GetInt("Level", 1);

Global Data API

Global data is shared across all profiles - perfect for settings and preferences.

Basic Types

// Integers
KPlayerPrefs.SetInt("GraphicsQuality", 2);
int quality = KPlayerPrefs.GetInt("GraphicsQuality", 1);

// Floats
KPlayerPrefs.SetFloat("MasterVolume", 0.8f);
float volume = KPlayerPrefs.GetFloat("MasterVolume", 1.0f);

// Strings
KPlayerPrefs.SetString("PlayerLanguage", "English");
string language = KPlayerPrefs.GetString("PlayerLanguage", "English");

// Booleans
KPlayerPrefs.SetBool("FullscreenMode", true);
bool fullscreen = KPlayerPrefs.GetBool("FullscreenMode", false);

Advanced Types

// Vector2 - Perfect for screen coordinates, 2D positions
Vector2 screenPos = new Vector2(Input.mousePosition.x, Input.mousePosition.y);
KPlayerPrefs.SetVector2("LastMousePosition", screenPos);
Vector2 savedPos = KPlayerPrefs.GetVector2("LastMousePosition", Vector2.zero);

// Vector3 - For 3D positions, rotations, scales

Security & Encryption

KPlayerPrefs automatically encrypts all data using industry-standard encryption.

Encryption Details

  • Algorithm: AES-256 in CBC mode
  • Key Derivation: PBKDF2 with 10,000 iterations
  • Salt: Fixed salt for consistency (customizable in source)
  • Padding: PKCS7 padding for block alignment

Password Management

  • Editor-Based Password Editing: Change encryption passwords safely through the KPlayerPrefs Editor
  • Secure Password Changes: When changing passwords, all existing data is automatically cleared to prevent corruption
  • Development-Only Access: Password management is restricted to the Unity Editor for security

Security Features

Event System

public class SaveSystemManager : MonoBehaviour
{
    void Start()
    {
        // Subscribe to data change events
        KPlayerPrefs.OnDataChanged += HandleDataChanged;
        KPlayerPrefs.OnActiveProfileChanged += HandleProfileChanged;
        KPlayerPrefs.OnProfileCreated += HandleProfileCreated;
        KPlayerPrefs.OnProfileDeleted += HandleProfileDeleted;
    }

    void OnDestroy()
    {
        // Unsubscribe to prevent memory leaks
        KPlayerPrefs.OnDataChanged -= HandleDataChanged;
        KPlayerPrefs.OnActiveProfileChanged -= HandleProfileChanged;
        KPlayerPrefs.OnProfileCreated -= HandleProfileCreated;
        KPlayerPrefs.OnProfileDeleted -= HandleProfileDeleted;
    }

    private void HandleDataChanged()
    {
        Debug.Log("KPlayerPrefs data has changed");
        // Update UI, sync to cloud, etc.
    }

    private void HandleProfileChanged(string newProfileName)
    {
        Debug.Log($"Active profile changed to: {newProfileName}");
        // Update profile-specific UI elements
        RefreshPlayerUI();
    }

    private void HandleProfileCreated(string profileName)
    {
        Debug.Log($"New profile created: {profileName}");
        // Update profile selection dropdown
        RefreshProfileDropdown();
    }

    private void HandleProfileDeleted(string profileName)
    {
        Debug.Log($"Profile deleted: {profileName}");
        // Clean up profile-specific resources
        RefreshProfileDropdown();
    }
}

Error Handling & Validation

public class SafeDataManager : MonoBehaviour
{
    public bool TrySwitchProfile(string profileName)
    {
        if (string.IsNullOrEmpty(profileName))
        {
            Debug.LogError("Profile name cannot be null or empty");
            return false;
        }

        if (!KPlayerPrefs.Profiles.Exists(profileName))
        {
            Debug.LogWarning($"Profile '{profileName}' does not exist");
            return false;
        }

        try
        {
            return KPlayerPrefs.Profiles.SetActive(profileName);
        }
        catch (System.Exception ex)
        {
            Debug.LogError($"Failed to switch to profile '{profileName}': {ex.Message}");
            return false;
        }
    }

    public bool SafeCreateProfile(string profileName)
    {
        // Validate profile name
        if (string.IsNullOrWhiteSpace(profileName))
        {
            Debug.LogError("Profile name cannot be empty or whitespace");
            return false;
        }

        if (profileName.IndexOfAny(Path.GetInvalidFileNameChars()) >= 0)
        {
            Debug.LogError("Profile name contains invalid characters");
            return false;
        }

        if (KPlayerPrefs.Profiles.Exists(profileName))
        {
            Debug.LogWarning($"Profile '{profileName}' already exists");
            return false;
        }

        try
        {
            return KPlayerPrefs.Profiles.Create(profileName);
        }
        catch (System.Exception ex)
        {
            Debug.LogError($"Failed to create profile '{profileName}': {ex.Message}");
            return false;
        }
    }

    public T SafeGetValue<T>(string key, T defaultValue, bool isProfile = false)
    {
        try
        {
            if (isProfile)
            {
                if (typeof(T) == typeof(int))
                    return (T)(object)KPlayerPrefs.Profiles.GetInt(key, (int)(object)defaultValue);
                if (typeof(T) == typeof(float))
                    return (T)(object)KPlayerPrefs.Profiles.GetFloat(key, (float)(object)defaultValue);
                if (typeof(T) == typeof(string))
                    return (T)(object)KPlayerPrefs.Profiles.GetString(key, (string)(object)defaultValue);
                if (typeof(T) == typeof(bool))
                    return (T)(object)KPlayerPrefs.Profiles.GetBool(key, (bool)(object)defaultValue);
            }
            else
            {
                if (typeof(T) == typeof(int))
                    return (T)(object)KPlayerPrefs.GetInt(key, (int)(object)defaultValue);
                if (typeof(T) == typeof(float))
                    return (T)(object)KPlayerPrefs.GetFloat(key, (float)(object)defaultValue);
                if (typeof(T) == typeof(string))
                    return (T)(object)KPlayerPrefs.GetString(key, (string)(object)defaultValue);
                if (typeof(T) == typeof(bool))
                    return (T)(object)KPlayerPrefs.GetBool(key, (bool)(object)defaultValue);
            }
        }
        catch (System.Exception ex)
        {
            Debug.LogError($"Failed to get value for key '{key}': {ex.Message}");
        }
        
        return defaultValue;
    }
}

Profile Information & Analytics

public class ProfileAnalytics : MonoBehaviour
{
    public void ShowProfileInfo(string profileName)
    {
        if (!KPlayerPrefs.Profiles.Exists(profileName)) return;

        // Get profile information
        var profileInfo = KPlayerPrefs.Profiles.GetInfo(profileName);
        Debug.Log($"Profile: {profileName}");
        Debug.Log($"Created: {profileInfo.creationDate:yyyy/MM/dd HH:mm:ss}");
        Debug.Log($"Last Access: {profileInfo.lastAccessDate:yyyy/MM/dd HH:mm:ss}");
        Debug.Log($"Data Size: {profileInfo.dataSize} bytes");
        Debug.Log($"Key Count: {profileInfo.keyCount}");

        // Get all data for analysis
        var accessor = KPlayerPrefs.Profiles.GetProfileAccessor(profileName);
        var allData = accessor.GetAllData();
        
        // Analyze data types
        int intCount = 0, floatCount = 0, stringCount = 0, boolCount = 0;
        int vector2Count = 0, vector3Count = 0, colorCount = 0, dateTimeCount = 0;

        foreach (var kvp in allData)
        {
            switch (kvp.Value)
            {
                case int _: intCount++; break;
                case float _: floatCount++; break;
                case string _: stringCount++; break;
                case bool _: boolCount++; break;
                case Vector2 _: vector2Count++; break;
                case Vector3 _: vector3Count++; break;
                case Color _: colorCount++; break;
                case DateTime _: dateTimeCount++; break;
            }
        }

        Debug.Log($"Data Types: {intCount} ints, {floatCount} floats, {stringCount} strings, " +
                 $"{boolCount} bools, {vector2Count} Vector2s, {vector3Count} Vector3s, " +
                 $"{colorCount} colors, {dateTimeCount} DateTimes");
    }

    public void ShowAllProfilesComparison()
    {
        string[] profiles = KPlayerPrefs.Profiles.GetAll();
        Debug.Log($"Total Profiles: {profiles.Length}");

        foreach (string profile in profiles)
        {
            var info = KPlayerPrefs.Profiles.GetInfo(profile);
            var accessor = KPlayerPrefs.Profiles.GetProfileAccessor(profile);
            
            int level = accessor.GetInt("Level", 0);
            float experience = accessor.GetFloat("Experience", 0f);
            DateTime lastPlayed = accessor.GetDateTime("LastPlayed", DateTime.MinValue);
            
            Debug.Log($"Profile '{profile}': Level {level}, XP {experience:F0}, " +
                     $"Last played {(DateTime.Now - lastPlayed).Days} days ago");
        }
    }
}

Optimization Tips

  1. Use Global Data for Settings: Store UI preferences, graphics settings globally
  2. Use Profile Data for Progress: Store player progress, inventory per profile
  3. Batch Operations: Group multiple Set operations together
  4. Cache Frequently Used Values: Don't call Get methods every frame

Performance Characteristics

  • Memory: Minimal RAM usage with lazy loading
  • Disk I/O: Efficient binary serialization with compression
  • Encryption Overhead: ~5-10% performance cost for security
  • Scalability: Handles thousands of keys efficiently

From Unity PlayerPrefs

Step 1: Add the using statement

using KairosoloSystems;

Step 2: Decide on data scope (Global vs Profile)

// Settings ‚Üí Global

// becomes
KPlayerPrefs.SetFloat("Volume", 0.8f);

// Player data ‚Üí Profile
PlayerPrefs.SetInt("Level", 5);
// becomes
KPlayerPrefs.Profiles.SetInt("Level", 5);

Step 3: Handle profile creation

void Start()

    }
}

Settings Manager

public class SettingsManager : MonoBehaviour
{
    [Header("Audio Settings")]
    public float masterVolume = 0.8f;
    public float sfxVolume = 0.8f;
    public float musicVolume = 0.8f;
    
    [Header("Graphics Settings")]
    public int qualityLevel = 2;
    public bool isFullscreen = true;
    public string resolution = "1920x1080";
    
    void Start()
    {
        LoadSettings();
        ApplySettings();
    }
    
    void LoadSettings()
    {
        // Load global settings
        masterVolume = KPlayerPrefs.GetFloat("Audio_MasterVolume", 0.8f);
        sfxVolume = KPlayerPrefs.GetFloat("Audio_SFXVolume", 0.8f);
        musicVolume = KPlayerPrefs.GetFloat("Audio_MusicVolume", 0.8f);
        
        qualityLevel = KPlayerPrefs.GetInt("Graphics_Quality", 2);
        isFullscreen = KPlayerPrefs.GetBool("Graphics_Fullscreen", true);
        resolution = KPlayerPrefs.GetString("Graphics_Resolution", "1920x1080");
    }
    
    void ApplySettings()
    {
        AudioListener.volume = masterVolume;
        QualitySettings.SetQualityLevel(qualityLevel);
        Screen.fullScreen = isFullscreen;
        
        // Parse and apply resolution
        var parts = resolution.Split('x');
        if (parts.Length == 2)
        {
            if (int.TryParse(parts[0], out int width) && int.TryParse(parts[1], out int height))
            {
                Screen.SetResolution(width, height, isFullscreen);
            }
        }
    }
    
    public void SaveSettings()
    {
        KPlayerPrefs.SetFloat("Audio_MasterVolume", masterVolume);
        KPlayerPrefs.SetFloat("Audio_SFXVolume", sfxVolume);
        KPlayerPrefs.SetFloat("Audio_MusicVolume", musicVolume);
        
        KPlayerPrefs.SetInt("Graphics_Quality", qualityLevel);
        KPlayerPrefs.SetBool("Graphics_Fullscreen", isFullscreen);
        KPlayerPrefs.SetString("Graphics_Resolution", resolution);
        
        Debug.Log("Settings saved!");
    }
}

Save Game System

public class SaveGameSystem : MonoBehaviour
{
    [System.Serializable]
    public class SaveData
    {
        public string playerName;
        public int level;
        public float experience;
        public Vector3 position;
        public Color playerColor;
        public DateTime lastPlayed;
        public List<string> inventory;
    }
    
    public SaveData currentSave;
    
    void Start()
    {
        CreateDefaultProfileIfNeeded();
        LoadGame();
    }
    
    void CreateDefaultProfileIfNeeded()
    {
        if (!KPlayerPrefs.Profiles.Exists("MainSave"))
        {
            KPlayerPrefs.Profiles.Create("MainSave");
            Debug.Log("Created new save profile");
        }
        KPlayerPrefs.Profiles.SetActive("MainSave");
    }
    
    void LoadGame()
    {
        currentSave = new SaveData
        {
            playerName = KPlayerPrefs.Profiles.GetString("PlayerName", "New Player"),
            level = KPlayerPrefs.Profiles.GetInt("Level", 1),
            experience = KPlayerPrefs.Profiles.GetFloat("Experience", 0f),
            position = KPlayerPrefs.Profiles.GetVector3("Position", Vector3.zero),
            playerColor = KPlayerPrefs.Profiles.GetColor("PlayerColor", Color.white),
            lastPlayed = KPlayerPrefs.Profiles.GetDateTime("LastPlayed", DateTime.Now),
            inventory = LoadInventory()
        };
        
        // Apply loaded data
        transform.position = currentSave.position;
        GetComponent<Renderer>().material.color = currentSave.playerColor;
        
        Debug.Log($"Game loaded: {currentSave.playerName}, Level {currentSave.level}");
    }
    
    public void SaveGame()
    {
        currentSave.position = transform.position;
        currentSave.lastPlayed = DateTime.Now;
        
        KPlayerPrefs.Profiles.SetString("PlayerName", currentSave.playerName);
        KPlayerPrefs.Profiles.SetInt("Level", currentSave.level);
        KPlayerPrefs.Profiles.SetFloat("Experience", currentSave.experience);
        KPlayerPrefs.Profiles.SetVector3("Position", currentSave.position);
        KPlayerPrefs.Profiles.SetColor("PlayerColor", currentSave.playerColor);
        KPlayerPrefs.Profiles.SetDateTime("LastPlayed", currentSave.lastPlayed);
        
        SaveInventory(currentSave.inventory);
        
        Debug.Log("Game saved!");
    }
    
    List<string> LoadInventory()
    {
        var inventory = new List<string>();
        int itemCount = KPlayerPrefs.Profiles.GetInt("InventoryCount", 0);
        
        for (int i = 0; i < itemCount; i++)
        {
            string item = KPlayerPrefs.Profiles.GetString($"InventoryItem_{i}", "");
            if (!string.IsNullOrEmpty(item))
            {
                inventory.Add(item);
            }
        }
        
        return inventory;
    }
    
    void SaveInventory(List<string> inventory)
    {
        KPlayerPrefs.Profiles.SetInt("InventoryCount", inventory.Count);
        
        for (int i = 0; i < inventory.Count; i++)
        {
            KPlayerPrefs.Profiles.SetString($"InventoryItem_{i}", inventory[i]);
        }
    }
    
    public void CreateNewSaveSlot(string slotName)
    {
        if (KPlayerPrefs.Profiles.Create(slotName))
        {
            KPlayerPrefs.Profiles.SetActive(slotName);
            
            // Initialize new save with default values
            currentSave = new SaveData
            {
                playerName = "New Player",
                level = 1,
                experience = 0f,
                position = Vector3.zero,
                playerColor = Color.white,
                lastPlayed = DateTime.Now,
                inventory = new List<string>()
            };
            
            SaveGame();
            Debug.Log($"Created new save slot: {slotName}");
        }
    }
}

Multi-Player Profile Manager

public class MultiPlayerManager : MonoBehaviour
{
    [System.Serializable]
    public class PlayerInfo
    {
        public string name;
        public int level;
        public Color color;
        public DateTime created;
        public DateTime lastPlayed;
    }
    
    public List<PlayerInfo> players = new List<PlayerInfo>();
    public string currentPlayer = "";
    
    void Start()
    {
        RefreshPlayerList();
    }
    
    public void RefreshPlayerList()
    {
        players.Clear();
        string[] profileNames = KPlayerPrefs.Profiles.GetAll();
        
        foreach (string profileName in profileNames)
        {
            var accessor = KPlayerPrefs.Profiles.GetProfileAccessor(profileName);
            var info = KPlayerPrefs.Profiles.GetInfo(profileName);
            
            players.Add(new PlayerInfo
            {
                name = accessor.GetString("PlayerName", profileName),
                level = accessor.GetInt("Level", 1),
                color = accessor.GetColor("PlayerColor", Color.white),
                created = info.creationDate,
                lastPlayed = info.lastAccessDate
            });
        }
        
        currentPlayer = KPlayerPrefs.Profiles.ActiveProfile;
    }
    
    public void SwitchToPlayer(string playerName)
    {
        if (KPlayerPrefs.Profiles.Exists(playerName))
        {
            KPlayerPrefs.Profiles.SetActive(playerName);
            currentPlayer = playerName;
            Debug.Log($"Switched to player: {playerName}");
        }
    }
    
    public void CreateNewPlayer(string playerName, Color playerColor)
    {
        if (KPlayerPrefs.Profiles.Create(playerName))
        {
            var accessor = KPlayerPrefs.Profiles.GetProfileAccessor(playerName);
            accessor.SetString("PlayerName", playerName);
            accessor.SetInt("Level", 1);
            accessor.SetFloat("Experience", 0f);
            accessor.SetColor("PlayerColor", playerColor);
            accessor.SetDateTime("Created", DateTime.Now);
            
            RefreshPlayerList();
            Debug.Log($"Created new player: {playerName}");
        }
    }
    
    public void DeletePlayer(string playerName)
    {
        if (playerName != "Default" && KPlayerPrefs.Profiles.Exists(playerName))
        {
            KPlayerPrefs.Profiles.Delete(playerName);
            
            // Switch to Default if we deleted the active player
            if (currentPlayer == playerName)
            {
                KPlayerPrefs.Profiles.SetActive("Default");
                currentPlayer = "Default";
            }
            
            RefreshPlayerList();
            Debug.Log($"Deleted player: {playerName}");
        }
    }
    
    public void ShowPlayerStats()
    {
        Debug.Log($"=== Player Statistics ===");
        foreach (var player in players)
        {
            Debug.Log($"{player.name}: Level {player.level}, " +
                     $"Created {player.created:yyyy/MM/dd}, " +
                     $"Last played {(DateTime.Now - player.lastPlayed).Days} days ago");
        }
    }
}
