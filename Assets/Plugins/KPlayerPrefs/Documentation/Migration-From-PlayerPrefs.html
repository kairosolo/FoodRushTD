Pattern 1: Mixed Data Manager

public class SaveManager : MonoBehaviour
{
    [Header("Global Settings")]
    public float masterVolume = 0.8f;
    public string gameLanguage = "English";
    public bool showTutorials = true;
    
    [Header("Player Data")]
    public string playerName = "Hero";
    public int playerLevel = 1;
    public Vector3 lastPosition = Vector3.zero;
    
    void Start()
    {
        LoadData();
    }
    
    void LoadData()
    {
        // Load global settings
        masterVolume = KPlayerPrefs.GetFloat("MasterVolume", 0.8f);
        gameLanguage = KPlayerPrefs.GetString("Language", "English");
        showTutorials = KPlayerPrefs.GetBool("ShowTutorials", true);
        
        // Ensure profile exists
        if (!KPlayerPrefs.Profiles.Exists("MainPlayer"))
        {
            KPlayerPrefs.Profiles.Create("MainPlayer");
        }
        KPlayerPrefs.Profiles.SetActive("MainPlayer");
        
        // Load player data
        playerName = KPlayerPrefs.Profiles.GetString("PlayerName", "Hero");
        playerLevel = KPlayerPrefs.Profiles.GetInt("Level", 1);
        lastPosition = KPlayerPrefs.Profiles.GetVector3("LastPosition", Vector3.zero);
    }
    
    public void SaveData()
    {
        // Save global settings
        KPlayerPrefs.SetFloat("MasterVolume", masterVolume);
        KPlayerPrefs.SetString("Language", gameLanguage);
        KPlayerPrefs.SetBool("ShowTutorials", showTutorials);
        
        // Save player data
        KPlayerPrefs.Profiles.SetString("PlayerName", playerName);
        KPlayerPrefs.Profiles.SetInt("Level", playerLevel);
        KPlayerPrefs.Profiles.SetVector3("LastPosition", lastPosition);
        KPlayerPrefs.Profiles.SetDateTime("LastPlayed", DateTime.Now);
    }
}

Pattern 2: Gradual Migration

public class LegacyDataMigration : MonoBehaviour
{
    void Start()
    {
        MigrateOldPlayerPrefs();
    }
    
    void MigrateOldPlayerPrefs()
    {
        // Check if we've already migrated
        if (KPlayerPrefs.GetBool("MigrationComplete", false))
            return;
            
        Debug.Log("Migrating old PlayerPrefs data...");
        
        // Migrate global settings
        if (PlayerPrefs.HasKey("MasterVolume"))
        {
            KPlayerPrefs.SetFloat("MasterVolume", PlayerPrefs.GetFloat("MasterVolume"));
            PlayerPrefs.DeleteKey("MasterVolume");
        }
        
        if (PlayerPrefs.HasKey("Language"))
        {
            KPlayerPrefs.SetString("Language", PlayerPrefs.GetString("Language"));
            PlayerPrefs.DeleteKey("Language");
        }
        
        // Create default profile and migrate player data
        if (!KPlayerPrefs.Profiles.Exists("MigratedPlayer"))
        {
            KPlayerPrefs.Profiles.Create("MigratedPlayer");
            KPlayerPrefs.Profiles.SetActive("MigratedPlayer");
        }
        
        if (PlayerPrefs.HasKey("PlayerName"))
        {
            KPlayerPrefs.Profiles.SetString("PlayerName", PlayerPrefs.GetString("PlayerName"));
            PlayerPrefs.DeleteKey("PlayerName");
        }
        
        if (PlayerPrefs.HasKey("Level"))
        {
            KPlayerPrefs.Profiles.SetInt("Level", PlayerPrefs.GetInt("Level"));
            PlayerPrefs.DeleteKey("Level");
        }
        
        // Mark migration as complete
        KPlayerPrefs.SetBool("MigrationComplete", true);
        KPlayerPrefs.SetDateTime("MigrationDate", DateTime.Now);
        
        Debug.Log("Migration complete!");
    }
}

Example 1: Settings Menu

public class SettingsMenu : MonoBehaviour
{
    [Header("UI References")]
    public Slider volumeSlider;
    public Dropdown qualityDropdown;
    public Toggle fullscreenToggle;
    
    void Start()
    {
        LoadSettings();
    }
    
    void LoadSettings()
    {
        // Load global settings (shared across all profiles)
        volumeSlider.value = KPlayerPrefs.GetFloat("MasterVolume", 0.8f);
        qualityDropdown.value = KPlayerPrefs.GetInt("QualityLevel", 2);
        fullscreenToggle.isOn = KPlayerPrefs.GetBool("Fullscreen", true);
        
        // Apply settings immediately
        AudioListener.volume = volumeSlider.value;
        QualitySettings.SetQualityLevel(qualityDropdown.value);
        Screen.fullScreen = fullscreenToggle.isOn;
    }
    
    public void OnVolumeChanged()
    {
        KPlayerPrefs.SetFloat("MasterVolume", volumeSlider.value);
        AudioListener.volume = volumeSlider.value;
    }
    
    public void OnQualityChanged()
    {
        KPlayerPrefs.SetInt("QualityLevel", qualityDropdown.value);
        QualitySettings.SetQualityLevel(qualityDropdown.value);
    }
    
    public void OnFullscreenChanged()
    {
        KPlayerPrefs.SetBool("Fullscreen", fullscreenToggle.isOn);
        Screen.fullScreen = fullscreenToggle.isOn;
    }
}

Example 2: Player Progress

public class PlayerProgress : MonoBehaviour
{
    [Header("Player Stats")]
    public string playerName;
    public int level;
    public float experience;
    public Vector3 spawnPosition;
    
    void Start()
    {
        LoadProgress();
    }
    
    void LoadProgress()
    {
        // Ensure profile exists
        if (!KPlayerPrefs.Profiles.Exists("MainPlayer"))
        {
            CreateNewPlayer();
        }
        else
        {
            KPlayerPrefs.Profiles.SetActive("MainPlayer");
            LoadPlayerData();
        }
    }
    
    void CreateNewPlayer()
    {
        KPlayerPrefs.Profiles.Create("MainPlayer");
        KPlayerPrefs.Profiles.SetActive("MainPlayer");
        
        // Set initial values
        playerName = "New Player";
        level = 1;
        experience = 0f;
        spawnPosition = Vector3.zero;
        
        SaveProgress();
    }
    
    void LoadPlayerData()
    {
        playerName = KPlayerPrefs.Profiles.GetString("PlayerName", "New Player");
        level = KPlayerPrefs.Profiles.GetInt("Level", 1);
        experience = KPlayerPrefs.Profiles.GetFloat("Experience", 0f);
        spawnPosition = KPlayerPrefs.Profiles.GetVector3("SpawnPosition", Vector3.zero);
        
        // Position the player
        transform.position = spawnPosition;
    }
    
    public void SaveProgress()
    {
        KPlayerPrefs.Profiles.SetString("PlayerName", playerName);
        KPlayerPrefs.Profiles.SetInt("Level", level);
        KPlayerPrefs.Profiles.SetFloat("Experience", experience);
        KPlayerPrefs.Profiles.SetVector3("SpawnPosition", transform.position);
        KPlayerPrefs.Profiles.SetDateTime("LastPlayed", DateTime.Now);
    }
    
    public void LevelUp()
    {
        level++;
        experience = 0f;
        SaveProgress();
        Debug.Log($"Level up! Now level {level}");
    }
}

‚ö†Ô∏è Important Notes

Data Compatibility

  • KPlayerPrefs uses separate files - Your old PlayerPrefs data is safe
  • No automatic migration - You need to manually migrate existing data
  • Different encryption - KPlayerPrefs data is encrypted, PlayerPrefs is not

Profile Management

  • Always create profiles before using profile data
  • Default profile exists automatically but create your own for better control
  • Active profile determines where profile data is saved

Best Practices

  • Use Global for settings that apply to all players
  • Use Profile for progress that's specific to each player
  • Always provide defaults when getting values
  • Handle profile creation in your initialization code
