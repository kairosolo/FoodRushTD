Data Structures

ProfileInfo

public struct ProfileInfo
{
    public string name;
    public DateTime creationDate;
    public DateTime lastAccessDate;
    public long dataSize;        // Size in bytes
    public int keyCount;         // Number of keys stored
    public bool isActive;        // Is this the active profile
}

PerformanceStats

public struct PerformanceStats
{
    public int SaveCount;              // Total save operations
    public int LoadCount;              // Total load operations
    public float AverageSaveTime;      // Average save time in seconds
    public float AverageLoadTime;      // Average load time in seconds
    public int ActiveProfileItemCount; // Items in active profile
    public int GlobalItemCount;        // Global data items
    public int TotalProfiles;          // Total number of profiles
    public string ActiveProfile;       // Current active profile name
    public long EstimatedMemoryUsage;  // Estimated memory usage in bytes
}

ProfileDataAccessor

Same methods as KPlayerPrefs.Profiles.* but for specific profile:

public class ProfileDataAccessor
{
    // All data type methods (Set/Get for int, float, string, bool, Vector2, Vector3, Color, DateTime)
    public int GetInt(string key, int defaultValue = 0);
    public void SetInt(string key, int value);
    // ... (same pattern for all data types)
    
    // Utility methods
    public bool HasKey(string key);
    public void DeleteKey(string key);
    public void DeleteKeys(string[] keys);
    public void DeleteAll();
    public string[] GetAllKeys();
    public Dictionary<string, object> GetAllData();
    public int GetKeyCount();
}

Quick Reference Examples

Basic Usage

// Global data
KPlayerPrefs.SetFloat("Volume", 0.8f);
float volume = KPlayerPrefs.GetFloat("Volume", 1.0f);

// Profile data (active profile)
KPlayerPrefs.Profiles.SetInt("Level", 25);
int level = KPlayerPrefs.Profiles.GetInt("Level", 1);

// Direct profile access
KPlayerPrefs.Profiles.Profile("Player1").SetString("Name", "Hero");
string name = KPlayerPrefs.Profiles.Profile("Player1").GetString("Name", "Unknown");

Profile Management

// Create and switch
KPlayerPrefs.Profiles.Create("Player1");
KPlayerPrefs.Profiles.SetActive("Player1");

// Check and get info
if (KPlayerPrefs.Profiles.Exists("Player1"))
{
    ProfileInfo info = KPlayerPrefs.Profiles.GetInfo("Player1");
}

// List all profiles
string[] profiles = KPlayerPrefs.Profiles.GetAll();

File Operations

// Export with full path
string exportPath = Path.Combine(Application.persistentDataPath, "backup.json");
KPlayerPrefs.ExportActiveProfile(exportPath);

// Import with full path
string importPath = Path.Combine(Application.dataPath, "../Saves/restore.json");
KPlayerPrefs.ImportActiveProfile(importPath, merge: true);
